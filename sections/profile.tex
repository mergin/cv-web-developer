% =====================
% Shared Profile Header
% =====================
% Renders a two-column header used by both cv.tex and cover-letter.tex.
% Left column: name + role.
% Right column: contact information aligned to the right.
\newcommand{\renderprofileheader}{%
  % Top horizontal rule above identity/contact block.
  \topruleline%
  % Vertical breathing room between top rule and content row.
  \vspace{0.7em}%

  % Left identity column.
  % Width split with the right column is 0.58 / 0.38 (plus hfill gap).
  \begin{minipage}[t]{0.58\linewidth}%
    % Ensure top alignment baseline for the minipage.
    \vspace{0pt}%
    % Primary identity line.
    {\fontsize{30}{34}\selectfont\MakeUppercase{\ProfileName}}\par%
    % Space between name and title.
    \vspace{0.18em}%
    % Professional title/subtitle.
    {\large \ProfileTitle}\par%
  \end{minipage}\hfill%

  % Right contact column.
  \begin{minipage}[t]{0.38\linewidth}%
    % Ensure top alignment baseline for the minipage.
    \vspace{0pt}%
    % Right-align contact lines and icons.
    \raggedleft%
    % Phone.
    \small \ProfilePhone\enspace \cvPhoneIcon\par%
    % Vertical rhythm between contact items.
    \vspace{0.30em}%
    % Email.
    \small \href{mailto:\ProfileEmail}{\ProfileEmail}\enspace \cvMailIcon\par%
    % Vertical rhythm between contact items.
    \vspace{0.30em}%
    % Location.
    \small \ProfileAddress\enspace \cvMapIcon\par%
    % Vertical rhythm between contact items.
    \vspace{0.30em}%
    % GitHub profile link.
    \small \href{https://\ProfileGitHub}{\ProfileGitHub}\enspace \cvGitHubIcon\par%
  \end{minipage}\par%

  % Space before lower divider line.
  \vspace{0.62em}%
  % Bottom horizontal rule beneath the header block.
  \noindent\color{ink}\rule{\linewidth}{0.55pt}\par%
}
